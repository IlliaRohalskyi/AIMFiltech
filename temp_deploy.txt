# Set variables
AWS_REGION=eu-central-1
AWS_ACCOUNT_ID=961341542251
REPO_NAME=sagemaker-ecr-repo
IMAGE_TAG=20250720-202135-8331172

# Create ECR repository (only needed once; safe to re-run)
aws ecr create-repository --repository-name $REPO_NAME --region $AWS_REGION || true

# Authenticate Docker to ECR
aws ecr get-login-password --region $AWS_REGION | \
  docker login --username AWS --password-stdin $AWS_ACCOUNT_ID.dkr.ecr.$AWS_REGION.amazonaws.com

# Build Docker image
docker build --no-cache -f Dockerfile.sagemaker -t $REPO_NAME:$IMAGE_TAG .

# Tag the image for ECR
docker tag $REPO_NAME:$IMAGE_TAG $AWS_ACCOUNT_ID.dkr.ecr.$AWS_REGION.amazonaws.com/$REPO_NAME:$IMAGE_TAG

# Push the image to ECR
docker push $AWS_ACCOUNT_ID.dkr.ecr.$AWS_REGION.amazonaws.com/$REPO_NAME:$IMAGE_TAG
